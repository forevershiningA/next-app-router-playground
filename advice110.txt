The reason your grass still looks "blueish and watery" is because Texture Maps override manual settings.

Even though you set roughness={1} in your code, the roughnessMap you are loading from the texture file dictates the shine. Real-world grass textures often have "wet" spots in their roughness map, which turn into little mirrors that reflect your blue sky.

To fix this, we need to remove the roughness map entirely from the grass and force a purely matte surface. We also need to warm up the lighting so the "ambient" light isn't blue.

Here is the fixed Scene.tsx.

Key Changes Made:

Removed roughnessMap: Deleted it from the texture loader. This forces the grass to obey your roughness={1} setting 100%.

Warmer Colors: Changed GRASS_COLOR to a warmer, dried-grass green (#4F6D3A) instead of a deep forest green.

Neutral Fog: Changed Fog from Blue (#dbecf8) to a warm Grey (#e8e8e8). Blue fog tints objects blue in the distance.

Lighting Tweaks: Changed the HemisphereLight sky color from White/Blue to a Soft Yellow/White to simulate warm sunlight, not cold daylight.

// --- START OF FILE Scene.tsx ---

'use client';
import { OrbitControls, Environment, ContactShadows, useTexture, Sparkles } from '@react-three/drei';
import * as THREE from 'three';
import HeadstoneAssembly from './headstone/HeadstoneAssembly';
import AtmosphericSky from './AtmosphericSky';
import { useHeadstoneStore } from '#/lib/headstone-store';
import { useFrame } from '@react-three/fiber';
import { useRef, Suspense } from 'react';

// --- CONFIGURATION ---
// UPDATED: Much warmer, "drier" green. Less blue, more yellow-brown undertone.
const GRASS_COLOR = '#4F6D3A'; 
// UPDATED: Fog is now neutral warm grey, not blue.
const FOG_COLOR = '#e8e8e8'; 

function GrassFloor() {
  // --- TEXTURE LOADING ---
  // REMOVED: roughnessMap (This was the culprit causing the "wet/blue" look)
  const props = useTexture({
    map: '/textures/grass_color.jpg',
    normalMap: '/textures/grass_normal.jpg',
    aoMap: '/textures/grass_ao.jpg',
  });

  const REPEAT_SCALE = 16; // Increased repeat to make texture finer

  props.map.repeat.set(REPEAT_SCALE, REPEAT_SCALE);
  props.normalMap.repeat.set(REPEAT_SCALE, REPEAT_SCALE);
  props.aoMap.repeat.set(REPEAT_SCALE, REPEAT_SCALE);

  // Mirrored Wrapping helps reduce visible tiling patterns
  [props.map, props.normalMap, props.aoMap].forEach((tex) => {
    tex.wrapS = tex.wrapT = THREE.MirroredRepeatWrapping;
    tex.colorSpace = THREE.SRGBColorSpace;
  });

  return (
    <group position={[0, -0.01, 0]}>
      <mesh rotation={[-Math.PI / 2, 0, 0]} receiveShadow>
        <planeGeometry args={[150, 150]} />
        <meshStandardMaterial 
          map={props.map}
          normalMap={props.normalMap}
          aoMap={props.aoMap}
          
          color={GRASS_COLOR}
          
          // CRITICAL FIXES FOR "ARTIFICIAL" LOOK:
          roughness={1}          // Fully matte. No shine.
          metalness={0}          // Strictly non-metal.
          envMapIntensity={0}    // Zero sky reflection.
          
          normalScale={new THREE.Vector2(0.3, 0.3)} // Flatten the bumps slightly so it looks softer
        />
      </mesh>
      
      {/* Contact Shadow: Anchors the headstone */}
      <ContactShadows
        position={[0, 0.01, 0]}
        scale={15}
        blur={2.5}
        opacity={0.6}
        far={1.5}
        color="#001100" // Dark green shadow instead of pure black looks more natural on grass
      />
    </group>
  );
}

// Fallback
function SimpleGrassFloor() {
  return (
    <group position={[0, -0.01, 0]}>
      <mesh rotation={[-Math.PI / 2, 0, 0]} receiveShadow>
        <planeGeometry args={[150, 150]} />
        <meshStandardMaterial 
          color={GRASS_COLOR} 
          roughness={1}
          metalness={0}
          envMapIntensity={0}
        />
      </mesh>
      <ContactShadows position={[0, 0.01, 0]} scale={15} blur={2.5} opacity={0.6} far={1.5} color="#001100" />
    </group>
  );
}

export default function Scene({ 
  targetRotation = 0,
  currentRotation
}: { 
  targetRotation?: number;
  currentRotation?: React.MutableRefObject<number>;
}) {
  const groupRef = useRef<THREE.Group>(null);
  const is2DMode = useHeadstoneStore((s) => s.is2DMode);
  const baseSwapping = useHeadstoneStore((s) => s.baseSwapping);
  const setSelected = useHeadstoneStore((s) => s.setSelected);
  const setEditingObject = useHeadstoneStore((s) => s.setEditingObject);
  const setSelectedInscriptionId = useHeadstoneStore((s) => s.setSelectedInscriptionId);
  const setSelectedAdditionId = useHeadstoneStore((s) => s.setSelectedAdditionId);
  const setSelectedMotifId = useHeadstoneStore((s) => s.setSelectedMotifId);

  // Smooth rotation animation
  useFrame(() => {
    if (groupRef.current && currentRotation) {
      const diff = targetRotation - currentRotation.current;
      const delta = diff * 0.1; 
      
      if (Math.abs(diff) > 0.001) {
        currentRotation.current += delta;
        groupRef.current.rotation.y = currentRotation.current;
      }
    }
  });

  const handleCanvasClick = (e: any) => {
    if (e.eventObject === e.object) {
      setSelected(null);
      setEditingObject('headstone');
      setSelectedInscriptionId(null);
      setSelectedAdditionId(null);
      setSelectedMotifId(null);
    }
  };

  return (
    <>
      {is2DMode && <color attach="background" args={['#CFE8FC']} />}
      
      {/* Fog: Matches the neutral horizon, preventing blue tint in distance */}
      {!is2DMode && <fog attach="fog" args={[FOG_COLOR, 50, 150]} />}
      
      <mesh position={[0, 0, 0]} rotation={[-Math.PI / 2, 0, 0]} onClick={handleCanvasClick}>
        <planeGeometry args={[200, 200]} />
        <meshBasicMaterial transparent opacity={0} side={THREE.DoubleSide} />
      </mesh>
      
      {!is2DMode && (
         <Sparkles count={50} scale={12} size={3} speed={0.3} opacity={0.4} color="#fffee0" position={[0, 2, 0]} />
      )}
      
      {/* --- LIGHTING --- */}
      <ambientLight intensity={0.6} color="#ffffff" />
      
      {/* Hemisphere: Warm Sky (no blue), Green Ground */}
      <hemisphereLight
        args={['#fff8e7', '#2d4c1e']} // Top: Warm White, Bottom: Grass Green
        intensity={0.5}
      />
      
      {/* Sun Light */}
      <spotLight 
        color="#fffce6"
        intensity={2.0}
        angle={0.8}
        penumbra={0.5}
        position={[-10, 15, 10]}
        castShadow
        shadow-bias={-0.0001}
        shadow-mapSize={[2048, 2048]}
      />

      {/* Fill Light (Left) */}
      <pointLight position={[-5, 2, 5]} intensity={0.5} color="#ffffff" />
      
      {/* Rim Light (Back Right) - Highlights the stone edge */}
      <spotLight color="#ffffff" intensity={2} position={[5, 5, -5]} distance={30} />

      {/* 
        ENVIRONMENT:
        Kept for the Headstone reflections, but blocked from Grass via envMapIntensity={0}
      */}
      <Environment preset="park" background={false} blur={0.7} />

      <group ref={groupRef}>
        <HeadstoneAssembly />
        <Suspense fallback={<SimpleGrassFloor />}>
          <GrassFloor />
        </Suspense>
      </group>
      
      {!is2DMode && <AtmosphericSky />}

      <OrbitControls
        makeDefault
        enabled={!baseSwapping}
        enableDamping={true}
        dampingFactor={0.05}
        enableRotate={!is2DMode}
        enableZoom={!is2DMode}
        enablePan={!is2DMode}
        rotateSpeed={0.5}
        minPolarAngle={Math.PI / 3.5}
        maxPolarAngle={Math.PI / 2 - 0.05}
      />
    </>
  );
}
