            <div className="flex w-full max-w-sm flex-col gap-10 z-20">
              
              {/* Stepper Navigation with Content Accordion */}
              <div className="flex flex-col gap-6 lg:border-l border-white/10 lg:pl-10">
                {steps.map((step) => {
                  const isActive = activeStep === step.id;
                  return (
                    <div key={step.id} className="flex flex-col items-start gap-4">
                      {/* Step Label */}
                      <button
                        onClick={() => setActiveStep(step.id)}
                        className={`text-left font-serif text-lg lg:text-3xl transition-all duration-300 ${
                          isActive 
                            ? 'text-[#F9F4E8] drop-shadow-[0_0_15px_rgba(255,255,255,0.3)]' 
                            : 'text-white/30 hover:text-white/60'
                        }`}
                      >
                        {step.label}
                      </button>

                      {/* Content: Shape Selection Grid */}
                      {isActive && step.id === 1 && (
                         <div className="grid grid-cols-4 gap-3 mt-2 animate-in fade-in slide-in-from-top-2 duration-300">
                           {shapes.map((shape, idx) => (
                             <button
                               key={shape.name}
                               onClick={() => setShapeIndex(idx)}
                               className={`relative w-14 h-14 rounded-lg flex items-center justify-center transition-all duration-200 ${
                                 shapeIndex === idx 
                                 ? 'bg-white/10 border-2 border-[#d4af37] shadow-[0_0_15px_rgba(212,175,55,0.3)]' 
                                 : 'bg-white/5 border border-white/10 hover:bg-white/10'
                               }`}
                               title={shape.name}
                             >
                               <div 
                                 className="w-8 h-8 bg-current"
                                 style={{
                                   maskImage: `url('${shape.file}')`,
                                   maskSize: "contain",
                                   maskRepeat: "no-repeat",
                                   maskPosition: "center",
                                   WebkitMaskImage: `url('${shape.file}')`,
                                   WebkitMaskSize: "contain",
                                   WebkitMaskRepeat: "no-repeat",
                                   WebkitMaskPosition: "center",
                                   backgroundColor: shapeIndex === idx ? '#F9F4E8' : 'rgba(255,255,255,0.5)'
                                 }}
                               />
                             </button>
                           ))}
                         </div>
                      )}

                      {/* Content: Material Selection Grid */}
                      {isActive && step.id === 2 && (
                         <div className="flex flex-wrap gap-4 mt-2 animate-in fade-in slide-in-from-top-2 duration-300">
                           {materials.map((mat, idx) => (
                              <button
                                key={mat.name}
                                onClick={() => setMaterialIndex(idx)}
                                className={`relative w-14 h-14 rounded-full overflow-hidden transition-all duration-300 ${
                                  materialIndex === idx 
                                  ? 'border-2 border-[#d4af37] ring-2 ring-[#d4af37]/20 scale-110' 
                                  : 'border border-white/30 hover:scale-105'
                                }`}
                                title={mat.name}
                              >
                                <Image src={mat.file} fill alt={mat.name} className="object-cover" />
                                {/* Glassy reflection */}
                                <div className="absolute inset-0 bg-gradient-to-tr from-transparent to-white/40 pointer-events-none" />
                              </button>
                           ))}
                         </div>
                      )}
                    </div>
                  );
                })}
              </div>

              {/* Action Buttons */}
              <div className="flex flex-col sm:flex-row gap-4 lg:pl-10 mt-4">
                <Link
                  href="/select-product"
                  className="rounded-full bg-gradient-to-r from-[#D4AF37] to-[#A67C00] border border-[#d4af37]/30 px-10 py-3.5 text-sm font-bold tracking-wider text-[#1a140f] shadow-lg transition-all hover:scale-105 hover:shadow-[0_0_20px_rgba(197,160,89,0.3)] hover:brightness-110 text-center"
                >
                  {activeStep < 4 ? 'Next' : 'Start Designing'}
                </Link>
                <Link
                  href="/designs"
                  className="rounded-full border border-white/20 bg-transparent px-10 py-3.5 text-sm font-bold tracking-wider text-white/80 hover:bg-white/5 hover:text-white hover:border-white/40 transition-all text-center"
                >
                  Browse Designs
                </Link>
              </div>

            </div>

        </div>

      </div>
    </div>
  );
}
